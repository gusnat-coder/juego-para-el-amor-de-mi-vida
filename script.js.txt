function crearCorazon(){

    const corazon = document.createElement("div");

    corazon.classList.add("heart");

    corazon.innerHTML = "❤️";

    corazon.style.left = Math.random() * 100 + "%";

    corazon.style.animationDuration = (6 + Math.random() * 5) + "s";

    corazon.style.fontSize = (15 + Math.random() * 25) + "px";

    document.getElementById("hearts").appendChild(corazon);

    setTimeout(() => {
        corazon.remove();
    }, 12000);

}

// Efecto máquina de escribir
function escribirTexto(texto){

    textoIntro.innerHTML="";

    let i=0;

    const velocidad=35;

    function escribir(){

        if(i<texto.length){

            textoIntro.innerHTML+=texto.charAt(i);

            i++;

            setTimeout(escribir,velocidad);

        }

    }

    escribir();

}
